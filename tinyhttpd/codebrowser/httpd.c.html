47/**********************************************************************/
48/* A request has caused a call to accept() on the server port to
49 * return. Process the request appropriately.
50 * Parameters: the socket connected to the client */
51/**********************************************************************/
52void* accept_request(void* tclient)
53{
54 int client = *(int*) tclient;
55 char buf[1024];
56 int numchars;
57 char method[255];
58 char url[255];
59 char path[512];
60 size_t i, j;
61 struct stat st;
62 int cgi = 0; /* becomes true if server decides this is a CGI
63 * program */
64 char *query_string = NULL;
65
66 numchars = get_line(client, buf, sizeof(buf));
67 i = 0; j = 0;
68 while (!ISspace(buf[j]) && (i < sizeof(method) - 1))
69 {
70 method[i] = buf[j];
71 i++; j++;
72 }
73 method[i] = '\0';
74
75 if (strcasecmp(method, "GET") && strcasecmp(method, "POST"))
76 {
77 unimplemented(client);
78 return NULL;
79 }
80
81 if (strcasecmp(method, "POST") == 0)
82 cgi = 1;
83
84 i = 0;
85 while (ISspace(buf[j]) && (j < sizeof(buf)))
86 j++;
87 while (!ISspace(buf[j]) && (i < sizeof(url) - 1) && (j < sizeof(buf)))
88 {
89 url[i] = buf[j];
90 i++; j++;
91 }
92 url[i] = '\0';
93
94 if (strcasecmp(method, "GET") == 0)
95 {
96 query_string = url;
97 while ((*query_string != '?') && (*query_string != '\0'))
98 query_string++;
99 if (*query_string == '?')
100 {
101 cgi = 1;
102 *query_string = '\0';
103 query_string++;
104 }
105 }
106
107 sprintf(path, "htdocs%s", url);
108 if (path[strlen(path) - 1] == '/')
109 strcat(path, "index.html");
110 if (stat(path, &st) == -1) {
111 while ((numchars > 0) && strcmp("\n", buf)) /* read & discard headers */
112 numchars = get_line(client, buf, sizeof(buf));
113 not_found(client);
114 }
115 else
116 {
117 if ((st.st_mode & S_IFMT) == S_IFDIR)
118 strcat(path, "/index.html");
119 if ((st.st_mode & S_IXUSR) ||
120 (st.st_mode & S_IXGRP) ||
121 (st.st_mode & S_IXOTH) )
122 cgi = 1;
123 if (!cgi)
124 serve_file(client, path);
125 else
126 execute_cgi(client, path, method, query_string);
127 }
128
129 close(client);
130 return NULL;
131}

133/**********************************************************************/
134/* Inform the client that a request it has made has a problem.
135 * Parameters: client socket */
136/**********************************************************************/
137void bad_request(int client)
138{
139 char buf[1024];
140
141 sprintf(buf, "HTTP/1.0 400 BAD REQUEST\r\n");
142 send(client, buf, sizeof(buf), 0);
143 sprintf(buf, "Content-type: text/html\r\n");
144 send(client, buf, sizeof(buf), 0);
145 sprintf(buf, "\r\n");
146 send(client, buf, sizeof(buf), 0);
147 sprintf(buf, "<P>Your browser sent a bad request, ");
148 send(client, buf, sizeof(buf), 0);
149 sprintf(buf, "such as a POST without a Content-Length.\r\n");
150 send(client, buf, sizeof(buf), 0);
151}

153/**********************************************************************/
154/* Put the entire contents of a file out on a socket. This function
155 * is named after the UNIX "cat" command, because it might have been
156 * easier just to do something like pipe, fork, and exec("cat").
157 * Parameters: the client socket descriptor
158 * FILE pointer for the file to cat */
159/**********************************************************************/
160void cat(int client, FILE *resource)
161{
162 char buf[1024];
163
164 fgets(buf, sizeof(buf), resource);
165 while (!feof(resource))
166 {
167 send(client, buf, strlen(buf), 0);
168 fgets(buf, sizeof(buf), resource);
169 }
170}

201/**********************************************************************/
202/* Execute a CGI script. Will need to set environment variables as
203 * appropriate.
204 * Parameters: client socket descriptor
205 * path to the CGI script */
206/**********************************************************************/
207void execute_cgi(int client, const char *path,
208 const char *method, const char *query_string)
209{
210 char buf[1024];
211 int cgi_output[2];
212 int cgi_input[2];
213 pid_t pid;
214 int status;
215 int i;
216 char c;
217 int numchars = 1;
218 int content_length = -1;
219
220 buf[0] = 'A'; buf[1] = '\0';
221 if (strcasecmp(method, "GET") == 0)
222 while ((numchars > 0) && strcmp("\n", buf)) /* read & discard headers */
223 numchars = get_line(client, buf, sizeof(buf));
224 else /* POST */
225 {
226 numchars = get_line(client, buf, sizeof(buf));
227 while ((numchars > 0) && strcmp("\n", buf))
228 {
229 buf[15] = '\0';
230 if (strcasecmp(buf, "Content-Length:") == 0)
231 content_length = atoi(&(buf[16]));
232 numchars = get_line(client, buf, sizeof(buf));
233 }
234 if (content_length == -1) {
235 bad_request(client);
236 return;
237 }
238 }
239
240 sprintf(buf, "HTTP/1.0 200 OK\r\n");
241 send(client, buf, strlen(buf), 0);
242
243 if (pipe(cgi_output) < 0) {
244 cannot_execute(client);
245 return;
246 }
247 if (pipe(cgi_input) < 0) {
248 cannot_execute(client);
249 return;
250 }
251
252 if ( (pid = fork()) < 0 ) {
253 cannot_execute(client);
254 return;
255 }
256 if (pid == 0) /* child: CGI script */
257 {
258 char meth_env[255];
259 char query_env[255];
260 char length_env[255];
261
262 dup2(cgi_output[1], 1);
263 dup2(cgi_input[0], 0);
264 close(cgi_output[0]);
265 close(cgi_input[1]);
266 sprintf(meth_env, "REQUEST_METHOD=%s", method);
267 putenv(meth_env);
268 if (strcasecmp(method, "GET") == 0) {
269 sprintf(query_env, "QUERY_STRING=%s", query_string);
270 putenv(query_env);
271 }
272 else { /* POST */
273 sprintf(length_env, "CONTENT_LENGTH=%d", content_length);
274 putenv(length_env);
275 }
276 execl(path, path, NULL);
277 exit(0);
278 } else { /* parent */
279 close(cgi_output[1]);
280 close(cgi_input[0]);
281 if (strcasecmp(method, "POST") == 0)
282 for (i = 0; i < content_length; i++) {
283 recv(client, &c, 1, 0);
284 write(cgi_input[1], &c, 1);
285 }
286 while (read(cgi_output[0], &c, 1) > 0)
287 send(client, &c, 1, 0);
288
289 close(cgi_output[0]);
290 close(cgi_input[1]);
291 waitpid(pid, &status, 0);
292 }
293}

295/**********************************************************************/
296/* Get a line from a socket, whether the line ends in a newline,
297 * carriage return, or a CRLF combination. Terminates the string read
298 * with a null character. If no newline indicator is found before the
299 * end of the buffer, the string is terminated with a null. If any of
300 * the above three line terminators is read, the last character of the
301 * string will be a linefeed and the string will be terminated with a
302 * null character.
303 * Parameters: the socket descriptor
304 * the buffer to save the data in
305 * the size of the buffer
306 * Returns: the number of bytes stored (excluding null) */
307/**********************************************************************/
308int get_line(int sock, char *buf, int size)
309{
310 int i = 0;
311 char c = '\0';
312 int n;
313
314 while ((i < size - 1) && (c != '\n'))
315 {
316 n = recv(sock, &c, 1, 0);
317 /* DEBUG printf("%02X\n", c); */
318 if (n > 0)
319 {
320 if (c == '\r')
321 {
322 n = recv(sock, &c, 1, MSG_PEEK);
323 /* DEBUG printf("%02X\n", c); */
324 if ((n > 0) && (c == '\n'))
325 recv(sock, &c, 1, 0);
326 else
327 c = '\n';
328 }
329 buf[i] = c;
330 i++;
331 }
332 else
333 c = '\n';
334 }
335 buf[i] = '\0';
336
337 return(i);
338}

387/**********************************************************************/
388/* Send a regular file to the client. Use headers, and report
389 * errors to client if they occur.
390 * Parameters: a pointer to a file structure produced from the socket
391 * file descriptor
392 * the name of the file to serve */
393/**********************************************************************/
394void serve_file(int client, const char *filename)
395{
396 FILE *resource = NULL;
397 int numchars = 1;
398 char buf[1024];
399
400 buf[0] = 'A'; buf[1] = '\0';
401 while ((numchars > 0) && strcmp("\n", buf)) /* read & discard headers */
402 numchars = get_line(client, buf, sizeof(buf));
403
404 resource = fopen(filename, "r");
405 if (resource == NULL)
406 not_found(client);
407 else
408 {
409 headers(client, filename);
410 cat(client, resource);
411 }
412 fclose(resource);
413}

415/**********************************************************************/
416/* This function starts the process of listening for web connections
417 * on a specified port. If the port is 0, then dynamically allocate a
418 * port and modify the original port variable to reflect the actual
419 * port.
420 * Parameters: pointer to variable containing the port to connect on
421 * Returns: the socket */
422/**********************************************************************/
423int startup(u_short *port)
424{
425 int httpd = 0;
426 struct sockaddr_in name;
427
428 httpd = socket(PF_INET, SOCK_STREAM, 0);
429 if (httpd == -1)
430 error_die("socket");
431 memset(&name, 0, sizeof(name));
432 name.sin_family = AF_INET;
433 name.sin_port = htons(*port);
434 name.sin_addr.s_addr = htonl(INADDR_ANY);
435 if (bind(httpd, (struct sockaddr *)&name, sizeof(name)) < 0)
436 error_die("bind");
437 if (*port == 0) /* if dynamically allocating a port */
438 {
439 socklen_t namelen = sizeof(name);
440 if (getsockname(httpd, (struct sockaddr *)&name, &namelen) == -1)
441 error_die("getsockname");
442 *port = ntohs(name.sin_port);
443 }
444 if (listen(httpd, 5) < 0)
445 error_die("listen");
446 return(httpd);
447}

478int main(void)
479{
480 int server_sock = -1;
481 u_short port = 0;
482 int client_sock = -1;
483 struct sockaddr_in client_name;
484 socklen_t client_name_len = sizeof(client_name);
485 pthread_t newthread;
486
487 server_sock = startup(&port);
488 printf("httpd running on port %d\n", port);
489
490 while (1)
491 {
492 client_sock = accept(server_sock,
493 (struct sockaddr *)&client_name,
494 &client_name_len);
495 if (client_sock == -1)
496 error_die("accept");
497 /* accept_request(client_sock); */
498 if (pthread_create(&newthread , NULL, accept_request, (void*)& client_sock) != 0)
499 perror("pthread_create");
500 }
501
502 close(server_sock);
503
504 return(0);
505}
